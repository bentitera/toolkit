def artist_search():
    base = 'https://search.azlyrics.com/search.php?q='
    artist = input('Who are you looking for?')
    for i in str.split(str.lower(artist)):
        base = base + '+' + str(i) + '&w=songs&p=1'
    r = requests.get(base)
    soup = BeautifulSoup(r.text, 'html.parser')
    return soup.url
