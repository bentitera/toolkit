def eighties_avg(filepath, country):
    df = pd.read_csv(filepath, index_col=0)
    c = df.loc[country]
    year = 1983
    gdp_decade = 'gdpPercap_' + str(year // 10)
    avg = (c.loc[gdp_decade + '2'] + c.loc[gdp_decade + '7']) / 2
    print('The average in the 80s for ', country, 'was', avg)
    return avg
